Sorting Algorithm

Bubble Sort

void bubbleSort(int* ary, unsigned len)
{
    for (unsigned i = 0; i<len-1; ++i) {
        for (unsigned j = 1; j < len; ++j) { // 每个循环最大的都会到低位
            if (ary[i] < ary[j])
            {
                ary[i] = ary[i] ^ ary[j];
                ary[j] = ary[i] ^ ary[j];
                ary[i] = ary[i] ^ ary[j];
            }
        }

    }

}
