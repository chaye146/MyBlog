Pre-order traversal

root->left->right 

void preorder(Node* root)
{
    if (!root)
    {
        return;
    }

    visit(root);
    preorder(root->left);
    preorder(root->right);
}

In-order traversal

left->root->right; 

void inorder(Node* root)
{
    if (!root)
    {
        return;
    }

    inorder(root->left);
    visit(root);
    inorder(root->right);
}

Post-order traversal

right->left->root 

void postorder(Node* root)
{
    if (!root)
    {
        return;
    }

    postorder(root->right);
    postorder(root->left);
    visit(root);
}
